# Gateway Implementation Plan

> **Status**: All tasks implemented.  See `docs/superpowers/specs/2026-06-26-gateway-design.md` for the final current design.
> 
> Key differences from this original plan:
> - Managers split into `_ai_manager.py` / `_session_manager.py` / `_manager.py` (shared socket helpers)
> - `AiCreateRequest` / `SessionCreateRequest` removed — `create()` uses plain parameters (server parses body directly)
> - `DeleteResponse` removed — `delete()` returns `None` (server constructs HTTP response inline)
> - `get_channel_socket` renamed to `get_socket` for consistency with AIManager
> - `"type": "file"` chunk removed from chat endpoint (security); only `text` / `blob` remain
> - ChatManager handles blob materialisation to `~/Downloads/.psi/<date>/` (persistent, not temp)
> - `GatewayTray` system tray icon support added (`_tray.py`, optional via `--tray`)
> - `Gateway.run()` now uses `async with create_task_group()` instead of manual `__aenter__/__aexit__`
> - Workspace `get_cwd()` and `GET /workspace/cwd` endpoint added
> - Files actually created: `_ai_manager.py`, `_session_manager.py`, `_chat_manager.py`, `_history_manager.py`, `_title_manager.py`, `_workspace_manager.py`, `_tray.py`, `_openapi.py`, `spa/` (frontend — out of scope here)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Gateway component that manages AI/Session lifecycle via OpenAPI REST and exposes a Web UI chat endpoint.

**Architecture:** Gateway is an aiohttp process. AI/Session run as anyio tasks spawned into a manually-managed TaskGroup (`__aenter__`/`__aexit__`). Each gets its own CancelScope for individual cancellation. Web UI chat endpoint reuses ChannelCore.

**Tech Stack:** aiohttp, anyio, tyro — all already in deps. Zero new dependencies.

**Note:** Socket paths follow platform convention: `/tmp/{socket_path}/...` on Linux (Unix socket), `\\.\pipe\{socket_path}\...` on Windows (Named Pipe). The `_socket_root()` module-level function handles this. Socket parent directories are created via `await anyio.Path(...).mkdir(parents=True, exist_ok=True)` — no `os.makedirs` or other synchronous IO.

```python
from __future__ import annotations

import sys
import uuid
from dataclasses import dataclass, field

import anyio
from loguru import logger

from psi_agent.ai import Ai
from psi_agent.session import Session


# --- Platform detection ---

def _socket_root(socket_path: str) -> str:
    if sys.platform == "win32":
        return fr"\\.\pipe\{socket_path}"
    return f"/tmp/{socket_path}"


# --- Request/Response types ---

@dataclass
class AiCreateRequest:
    provider: str
    model: str
    api_key: str
    base_url: str
    id: str = ""


@dataclass
class AiInfo:
    id: str
    socket: str
    provider: str
    model: str


@dataclass
class SessionCreateRequest:
    ai_id: str
    workspace: str
    id: str = ""


@dataclass
class SessionInfo:
    id: str
    ai_id: str
    workspace: str
    channel_socket: str


@dataclass
class DeleteResponse:
    id: str
    status: str = "stopped"


# --- Helpers ---

def _new_uuid() -> str:
    return uuid.uuid4().hex


async def _wait_socket(path: str, timeout_sec: float = 10.0) -> None:
    deadline = anyio.current_time() + timeout_sec
    sock = anyio.Path(path)
    while anyio.current_time() < deadline:
        if await sock.exists():
            await anyio.sleep(0.3)
            return
        await anyio.sleep(0.1)
    raise TimeoutError(f"Socket '{path}' not created within {timeout_sec}s")


# --- AIManager ---

@dataclass
class _AiEntry:
    scope: anyio.CancelScope
    socket: str
    provider: str
    model: str


@dataclass
class AIManager:
    _prefix: str
    _tg: anyio.TaskGroup
    _entries: dict[str, _AiEntry] = field(default_factory=dict)
    _lock: anyio.Lock = field(default_factory=anyio.Lock)

    def _socket_path(self, ai_id: str) -> str:
        return f"{_socket_root(self._prefix)}/ais/{ai_id}.sock"

    async def create(self, req: AiCreateRequest) -> AiInfo:
        ai_id = req.id or _new_uuid()
        async with self._lock:
            if ai_id in self._entries:
                raise ValueError(f"AI '{ai_id}' already exists")
            socket = self._socket_path(ai_id)
            await anyio.Path(socket).parent.mkdir(parents=True, exist_ok=True)
            ai = Ai(
                session_socket=socket,
                provider=req.provider,
                model=req.model,
                api_key=req.api_key,
                base_url=req.base_url,
            )
            scope = anyio.CancelScope()

            async def _run_ai() -> None:
                with scope:
                    await ai.run()

            self._tg.start_soon(_run_ai)
            self._entries[ai_id] = _AiEntry(scope=scope, socket=socket, provider=req.provider, model=req.model)
            await _wait_socket(socket)
            logger.info(f"AI '{ai_id}' created on {socket}")
            return AiInfo(id=ai_id, socket=socket, provider=req.provider, model=req.model)

    async def delete(self, ai_id: str) -> DeleteResponse:
        async with self._lock:
            if ai_id not in self._entries:
                raise LookupError(f"AI '{ai_id}' not found")
            entry = self._entries.pop(ai_id)
            entry.scope.cancel()
            logger.info(f"AI '{ai_id}' deleted")
            return DeleteResponse(id=ai_id)

    async def list(self) -> list[AiInfo]:
        return [
            AiInfo(id=aid, socket=e.socket, provider=e.provider, model=e.model)
            for aid, e in self._entries.items()
        ]

    def get_socket(self, ai_id: str) -> str:
        if ai_id not in self._entries:
            raise LookupError(f"AI '{ai_id}' not found")
        return self._entries[ai_id].socket

    def has(self, ai_id: str) -> bool:
        return ai_id in self._entries


# --- SessionManager ---

@dataclass
class _SessionEntry:
    scope: anyio.CancelScope
    channel_socket: str
    ai_id: str
    workspace: str


@dataclass
class SessionManager:
    _aim: AIManager
    _prefix: str
    _tg: anyio.TaskGroup
    _entries: dict[str, _SessionEntry] = field(default_factory=dict)
    _lock: anyio.Lock = field(default_factory=anyio.Lock)

    def _socket_path(self, session_id: str) -> str:
        return f"{_socket_root(self._prefix)}/sessions/{session_id}.sock"

    async def create(self, req: SessionCreateRequest) -> SessionInfo:
        if not self._aim.has(req.ai_id):
            raise LookupError(f"AI '{req.ai_id}' not found")
        session_id = req.id or _new_uuid()
        async with self._lock:
            if session_id in self._entries:
                raise ValueError(f"Session '{session_id}' already exists")
            ai_socket = self._aim.get_socket(req.ai_id)
            channel_socket = self._socket_path(session_id)
            await anyio.Path(channel_socket).parent.mkdir(parents=True, exist_ok=True)
            sess = Session(
                workspace=req.workspace,
                channel_socket=channel_socket,
                ai_socket=ai_socket,
                session_id=session_id if req.id else None,
            )
            scope = anyio.CancelScope()

            async def _run_session() -> None:
                with scope:
                    await sess.run()

            self._tg.start_soon(_run_session)
            self._entries[session_id] = _SessionEntry(
                scope=scope, channel_socket=channel_socket, ai_id=req.ai_id, workspace=req.workspace,
            )
            await _wait_socket(channel_socket)
            logger.info(f"Session '{session_id}' created on {channel_socket} -> AI '{req.ai_id}'")
            return SessionInfo(id=session_id, ai_id=req.ai_id, workspace=req.workspace, channel_socket=channel_socket)

    async def delete(self, session_id: str) -> DeleteResponse:
        async with self._lock:
            if session_id not in self._entries:
                raise LookupError(f"Session '{session_id}' not found")
            entry = self._entries.pop(session_id)
            entry.scope.cancel()
            logger.info(f"Session '{session_id}' deleted")
            return DeleteResponse(id=session_id)

    async def list(self) -> list[SessionInfo]:
        return [
            SessionInfo(id=sid, ai_id=e.ai_id, workspace=e.workspace, channel_socket=e.channel_socket)
            for sid, e in self._entries.items()
        ]

    def get_channel_socket(self, session_id: str) -> str:
        if session_id not in self._entries:
            raise LookupError(f"Session '{session_id}' not found")
        return self._entries[session_id].channel_socket

    def has(self, session_id: str) -> bool:
        return session_id in self._entries
```

```python
from __future__ import annotations

import json
from dataclasses import asdict

from aiohttp import web

from psi_agent.gateway._manager import (
    AiCreateRequest,
    AIManager,
    SessionCreateRequest,
    SessionManager,
)


def _json(data: object, status: int = 200) -> web.Response:
    return web.Response(
        text=json.dumps(data),
        content_type="application/json",
        status=status,
    )


def _error(message: str, status: int) -> web.Response:
    return _json({"error": message}, status=status)


def create_app(aim: AIManager, sm: SessionManager) -> web.Application:
    app = web.Application()
    app["aim"] = aim
    app["sm"] = sm

    app.router.add_post("/ais", _create_ai)
    app.router.add_delete("/ais/{ai_id}", _delete_ai)
    app.router.add_get("/ais", _list_ais)
    app.router.add_post("/sessions", _create_session)
    app.router.add_delete("/sessions/{session_id}", _delete_session)
    app.router.add_get("/sessions", _list_sessions)

    return app


async def _create_ai(request: web.Request) -> web.Response:
    aim: AIManager = request.app["aim"]
    try:
        body = await request.json()
        req = AiCreateRequest(**body)
        info = await aim.create(req)
        return _json(asdict(info), status=201)
    except (TypeError, ValueError) as e:
        return _error(str(e), status=400)
    except Exception as e:
        return _error(str(e), status=500)


async def _delete_ai(request: web.Request) -> web.Response:
    aim: AIManager = request.app["aim"]
    ai_id = request.match_info["ai_id"]
    try:
        info = await aim.delete(ai_id)
        return _json(asdict(info))
    except LookupError as e:
        return _error(str(e), status=404)
    except Exception as e:
        return _error(str(e), status=500)


async def _list_ais(request: web.Request) -> web.Response:
    aim: AIManager = request.app["aim"]
    return _json([asdict(i) for i in await aim.list()])


async def _create_session(request: web.Request) -> web.Response:
    sm: SessionManager = request.app["sm"]
    try:
        body = await request.json()
        req = SessionCreateRequest(**body)
        info = await sm.create(req)
        return _json(asdict(info), status=201)
    except (TypeError, ValueError, LookupError) as e:
        return _error(str(e), status=400)
    except Exception as e:
        return _error(str(e), status=500)


async def _delete_session(request: web.Request) -> web.Response:
    sm: SessionManager = request.app["sm"]
    session_id = request.match_info["session_id"]
    try:
        info = await sm.delete(session_id)
        return _json(asdict(info))
    except LookupError as e:
        return _error(str(e), status=404)
    except Exception as e:
        return _error(str(e), status=500)


async def _list_sessions(request: web.Request) -> web.Response:
    sm: SessionManager = request.app["sm"]
    return _json([asdict(i) for i in await sm.list()])
```

```python
from __future__ import annotations

from dataclasses import dataclass

import anyio
from aiohttp import web
from loguru import logger

from psi_agent._logging import setup_logging
from psi_agent._socket import create_site
from psi_agent.gateway._manager import AIManager, SessionManager
from psi_agent.gateway.server import create_app


@dataclass
class Gateway:
    listen: str = "http://0.0.0.0:8080"
    socket_path: str = "psi"
    verbose: bool = False

    async def run(self) -> None:
        setup_logging(verbose=self.verbose)

        tg = anyio.create_task_group()
        await tg.__aenter__()

        aim = AIManager(_prefix=self.socket_path, _tg=tg)
        sm = SessionManager(_aim=aim, _prefix=self.socket_path, _tg=tg)

        app = create_app(aim, sm)
        runner = web.AppRunner(app)
        await runner.setup()
        site = create_site(runner, self.listen)
        await site.start()
        logger.info(f"Gateway listening on {self.listen}")

        try:
            await anyio.sleep_forever()
        finally:
            logger.info("Shutting down Gateway")
            await runner.cleanup()
            await tg.__aexit__(None, None, None)
```

```python
from __future__ import annotations

import json

from aiohttp import web
from loguru import logger

from psi_agent.channel._core import ChannelCore
from psi_agent.channel._types import Chunk, FileChunk, TextChunk


async def handle_chat(request: web.Request) -> web.StreamResponse:
    sm = request.app["sm"]
    session_id = request.match_info["session_id"]

    try:
        channel_socket = sm.get_channel_socket(session_id)
    except LookupError:
        return _json_error(404, f"Session '{session_id}' not found")

    try:
        body = await request.json()
        raw_chunks: list[dict] = body.get("chunks", [])
        chunks: list[Chunk] = []
        for c in raw_chunks:
            if c.get("type") == "text":
                chunks.append(TextChunk(text=c["text"]))
            elif c.get("type") == "file":
                chunks.append(FileChunk(path=c["path"]))
    except Exception as e:
        return _json_error(400, f"Invalid request body: {e}")

    resp = web.StreamResponse(
        status=200,
        reason="OK",
        headers={"Content-Type": "text/event-stream"},
    )
    await resp.prepare(request)

    try:
        async with ChannelCore(session_socket=channel_socket, interval=0.0) as core:
            async for chunk in core.post(chunks):
                if isinstance(chunk, TextChunk):
                    await _write_sse(resp, {"type": "text", "text": chunk.text})
                elif isinstance(chunk, FileChunk):
                    await _write_sse(resp, {"type": "file", "path": chunk.path})
    except Exception as e:
        logger.debug(f"Chat error for session '{session_id}': {e}")
        await _write_sse(resp, {"type": "error", "error": str(e)})

    await _write_done(resp)
    return resp


async def _write_sse(resp: web.StreamResponse, data: dict) -> None:
    await resp.write(f"data: {json.dumps(data)}\n\n".encode())


async def _write_done(resp: web.StreamResponse) -> None:
    await resp.write(b"data: [DONE]\n\n")


def _json_error(status: int, message: str) -> web.Response:
    return web.Response(
        text=json.dumps({"error": message}),
        content_type="application/json",
        status=status,
    )
```
